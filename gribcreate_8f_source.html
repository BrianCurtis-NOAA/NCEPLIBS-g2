102 ! 1-4 of section 1.
103 !
104 lensec1=(iofst-ibeg)/8

106!
107! Put current byte total of message into Section 0
108!
109 call g2_sbytec(cgrib,zero,64,32)
110 call g2_sbytec(cgrib,lensec0+lensec1,96,32)
111
